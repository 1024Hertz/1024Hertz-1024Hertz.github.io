tags:

---

# 搜索镜像
~~~shell
docker search registry
~~~

# 安装镜像
~~~shell
docker pull registry
~~~

# 启动镜像
~~~shell
docker run -d \
    --restart=always \
    --name=registry \
    -p 5000:5000 \
    registry
~~~

# 添加信任地址
~~~shell
sudo vi /etc/docker/daemon.json
{
    # 让docker信任YourIP:5000 registry私服地址
    "insecure-registries":["YourIP:5000"]
}
~~~
